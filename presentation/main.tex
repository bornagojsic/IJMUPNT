\begin{frame}{Povezanost P9 i P10}
  \begin{block}{Interpretacijsko pitanje}
    Jesu li ispitanici koji ističu određenu prednost SMR tehnologije skloniji podržati državne subvencije?
  \end{block}
  \vspace{0.4cm}
  \begin{itemize}
    \item Povezanost P9 i P10 najizraženiji je statistički nalaz analize.
    \item Ispitanici koji prepoznaju sigurnosne značajke češće podržavaju subvencije.
    \item Ispitanici koji za P9 odgovaraju \textit{Ne znam} često nisu sigurni ni oko opravdanosti subvencija.
    \item Nalaz upućuje na vezu između jasne percepcije koristi SMR tehnologije i potpore javnom ulaganju.
  \end{itemize}
\end{frame}

\begin{frame}{Rasprava}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Tumačenje rezultata}
        \begin{itemize}
          \item SMR se ne promatra primarno kroz troškove.
          \item Najvažniji su sigurnost i izvedivost tehnologije.
          \item Potpora subvencijama postoji, ali uz visok udio nesigurnih ispitanika.
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Ograničenja}
        \begin{itemize}
          \item Anketa mjeri stavove, a ne tehničku ili ekonomsku opravdanost.
          \item Uzorak je dobno neuravnotežen prema mlađim ispitanicima.
          \item Rijetke kategorije, posebno kod županija, smanjuju pouzdanost dijela usporedbi.
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Zaključak}
  \begin{columns}[T]
    \begin{column}{0.56\textwidth}
      \begin{block}{Ključne poruke}
        \begin{itemize}
          \item Najčešće prepoznata prednost SMR tehnologije su poboljšane sigurnosne značajke (33,86\%).
          \item Većina ispitanika podržava državne subvencije za razvoj SMR tehnologije (59,01\%).
          \item Značajan udio odgovora \textit{Ne znam} pokazuje potrebu za dodatnim informiranjem javnosti.
          \item Podrška subvencijama povezana je s prepoznavanjem konkretnih prednosti SMR tehnologije.
        \end{itemize}
      \end{block}
    \end{column}
